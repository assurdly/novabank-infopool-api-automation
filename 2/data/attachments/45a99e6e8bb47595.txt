var jsonData = pm.response.json();

pm.test("Status code is 201", function () {
    pm.response.to.have.status(201);
});

pm.test("Reponse body contains standard success fields", function () {
    pm.expect(jsonData.status).to.eql(true);
    pm.expect(jsonData.response[0].batchId).is.not.null;
    pm.expect(jsonData.response[0].email).is.not.null;
    pm.expect(jsonData.response[0].status).is.not.null;
});

pm.test("The error message about the empty email is returned", function () {
    pm.expect(jsonData.response[0].message).to.eql('Invitation successfully sent.');
});
