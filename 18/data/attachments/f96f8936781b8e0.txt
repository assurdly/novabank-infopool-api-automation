var jsonData = pm.response.json();

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Reponse body contains success fields", function () {
    pm.expect(jsonData.status).to.eql(true);
    pm.expect(jsonData.response.data[0].comment).is.not.null;
    pm.expect(jsonData.response.data[0].created).is.not.null;
});
pm.environment.set("comment_id", jsonData.response.data[0].id);
