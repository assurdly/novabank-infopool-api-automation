var jsonData = pm.response.json();

pm.test("Status code is 404", function () {
    pm.response.to.have.status(404);
});
pm.test("Reponse body contains standard error fields", function () {
    pm.expect(jsonData.statusCode).to.eql(404);
    pm.expect(jsonData.path).is.not.null;
    pm.expect(jsonData.status).to.eql(false);
   pm.expect(jsonData.message).to.eql('Role with ID 0 not found'); 
});
