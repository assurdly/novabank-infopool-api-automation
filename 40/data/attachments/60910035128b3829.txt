var jsonData = pm.response.json();

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
    pm.expect(jsonData.status).to.eql(true);
});

pm.test("Activities are returned", function () {
    pm.expect(jsonData.response.data[0].created_on).to.not.equal("");
    pm.expect(jsonData.response.data[0].created_by).to.not.equal("");
    pm.expect(jsonData.response.data[0].modified_on).to.not.equal("");
    pm.expect(jsonData.response.data[0].event_id).to.not.equal("");
    pm.expect(jsonData.response.data[0].id).to.not.equal("");
    pm.expect(jsonData.response.data[0].entity_type).to.not.equal("");
    pm.expect(jsonData.response.data[0].description).to.not.equal("");
    pm.expect(jsonData.response.data[0].ip_address).to.not.equal("");
     pm.expect(jsonData.response.data[0].location).to.not.equal("");
});
