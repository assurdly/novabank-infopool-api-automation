var jsonData = pm.response.json();

pm.test("Status code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Response body contains standard success fields", function () {
    pm.expect(jsonData.status).to.eql(false);
    pm.expect(jsonData.statusCode).to.eql(400)
});
pm.test("The error message about the invalid action is returned", function () {
    pm.expect(jsonData.message[0]).to.eql('action must be one of the followings: update,delete,deactivate,activate');
});
