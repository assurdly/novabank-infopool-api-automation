var jsonData = pm.response.json();

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response body contains success fields", function () {
    pm.expect(jsonData.status).to.eql(true);
    pm.expect(jsonData.response.data.transaction_id).to.not.be.null;
    pm.expect(jsonData.response.data.transaction_date).to.not.be.null;
    pm.expect(jsonData.response.data.part_transaction_serial_number).to.not.be.null;
    pm.expect(jsonData.response.data.reference_number).to.not.be.null;
    pm.expect(jsonData.response.data.transaction_amount).to.not.be.null;
});
