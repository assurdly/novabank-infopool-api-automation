var jsonData = pm.response.json();

pm.test("Status code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Response body contains error fields", function () {
    pm.expect(jsonData.statusCode).to.eql(400);
    pm.expect(jsonData.message[0]).to.eql("priority must be one of the following values: LOW, MEDIUM, HIGH, CRITICAL");
});
