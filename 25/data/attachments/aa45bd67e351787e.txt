var jsonData = pm.response.json();

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response body contains success fields", function () {
    pm.expect(jsonData.status).to.eql(true);
    pm.expect(jsonData.response.data[0].customer_name).to.not.be.null;
    pm.expect(jsonData.response.data[0].customer_start_date).to.not.be.null;
    pm.expect(jsonData.response.data[0].phone).to.not.be.null;
    pm.expect(jsonData.response.data[0].type).to.not.be.null;
});
