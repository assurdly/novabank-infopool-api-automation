var jsonData = pm.response.json();

pm.test("Status code is 201", function () {
    pm.response.to.have.status(201);
});

pm.test("Response body contains success fields", function () {
    pm.expect(jsonData.status).to.eql(true);
    pm.expect(jsonData.response.created_by).to.not.be.null;
    pm.expect(jsonData.response.created_on).to.not.be.null;
    pm.expect(jsonData.response.comment).to.not.be.null;
});
