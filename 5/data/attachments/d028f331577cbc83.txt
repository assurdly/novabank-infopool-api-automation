var jsonData = pm.response.json();

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});
pm.test("Reponse body contains standard success fields", function () {
    pm.expect(jsonData.status).to.eql(true);
    pm.expect(jsonData.message).to.eql('Logout successful');
});

pm.test("Successful logged out message is returned", function () {
    pm.expect(jsonData.response.resonse.message).to.eql('Logged out successfully');
});
