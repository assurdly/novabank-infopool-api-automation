var jsonData = pm.response.json();

pm.test("Status code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Response body contains standard Error fields", function () {
    pm.expect(jsonData.error).to.eql('Bad Request');
    pm.expect(jsonData.statusCode).to.eql(400)
});
pm.test("The error message about the invalid id  is returned", function () {
    pm.expect(jsonData.message).to.eql('Validation failed (numeric string is expected)');
});
