var jsonData = pm.response.json();

pm.test("Status code is 201", function () {
    pm.response.to.have.status(201);
    pm.expect(jsonData.status).to.eql(true);
});

pm.test("Access credentials are returned", function () {
    pm.expect(jsonData.response.token_type).is.not.empty;
    pm.expect(jsonData.response.access_token).is.not.empty;
    pm.expect(jsonData.response.refresh_token).is.not.empty;
    pm.expect(jsonData.response.session_id).is.not.empty;
     pm.expect(jsonData.response.device_id).is.not.empty;

});
pm.environment.set("Bearer_token", jsonData.response.access_token);
pm.environment.set("refresh_token", jsonData.response.refresh_token);
