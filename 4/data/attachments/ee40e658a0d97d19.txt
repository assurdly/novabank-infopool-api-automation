var jsonData = pm.response.json();

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response body contains standard success fields", function () {
    pm.expect(jsonData.status).to.eql(false);
    pm.expect(jsonData.error.statusCode).to.eql(404)
});
pm.test("The error message about the invalid user is returned", function () {
    pm.expect(jsonData.error.message).to.eql('User not found');
});
