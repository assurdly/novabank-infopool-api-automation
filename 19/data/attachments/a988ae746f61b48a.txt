var jsonData = pm.response.json();

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Reponse body contains success fields", function () {
    pm.expect(jsonData.status).to.eql(true);
    pm.expect(jsonData.response.slug).to.not.be.oneOf([null, ""]);
    pm.expect(jsonData.response.name).to.not.be.oneOf([null, ""]);
    pm.expect(jsonData.response.parent).to.not.be.oneOf([null, ""]);
});
