var jsonData = pm.response.json();

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Reponse body contains success fields", function () {
    pm.expect(jsonData.status).to.eql(true);
    pm.expect(jsonData.response.uuid).to.not.be.null;
    pm.expect(jsonData.response.name).to.not.be.null;
    pm.expect(jsonData.response.description).to.not.be.null;
    pm.expect(jsonData.response.meta).to.not.be.empty;
})
