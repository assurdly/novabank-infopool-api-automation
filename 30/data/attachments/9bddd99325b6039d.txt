var jsonData = pm.response.json();

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response body contains success fields", function () {
    pm.expect(jsonData.status).to.eql(true);
    pm.expect(jsonData.response.data[0].account_id).to.not.be.null;
    pm.expect(jsonData.response.data[0].account_number).to.not.be.null;
    pm.expect(jsonData.response.data[0].scheme_code).to.not.be.null;
    pm.expect(jsonData.response.data[0].account_open_date).to.not.be.null;
    pm.expect(jsonData.response.data[0].ledger_balance).to.not.be.null;
});
